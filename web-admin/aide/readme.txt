// --------------------------------------------------------------------------------
// ----   Paramètres à connaître

$_GET['act']
	lst : une liste
	add : une insertion
	upd : une modification
	del : une suppression
	oth : une autre action

$_GET['sec']
	texte : les textes (billets, plans, articles)
	commentaire : les commentaires
	message : les messages envoyés par le formulaire
	parametre : les autres paramètres
	rss : les flux rss

$_GET['aff'] : affichage

$_GET['id'] : numéro d'identifiant
